#version 400

layout(location=0) in vec4 pos;
layout(location=1) in vec3 norm;
layout(location=2) in vec2 uv;

uniform float scale;
uniform mat4 rot;
uniform mat4 trans;
uniform mat4 persp;

out vec2 texcoord;
out vec4 n;

void main(void)
{
	gl_Position =  persp * trans * rot * vec4(pos.xyz * scale, 1.0); 
	texcoord = uv;
	n = -trans * rot * vec4(norm, 0.0);
}
